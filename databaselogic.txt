CREATE TABLE status(
  statusCode varchar2(10) NOT NULL UNIQUE,
  status varchar2(10) NOT NULL,
  date DATE,
  share INT(),
  PRIMARY KEY (statusCode)
  )
CREATE table permissions(
  statusCode varchar2(10) NOT NULL UNIQUE,
  like BIT(),
  comment BIT(),
  share BIT(),
  FOREIGN KEY (statusCode) REFERENCES status(statusCode)
)
